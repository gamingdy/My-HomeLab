import { Code } from "@astrojs/starlight/components";

Create a folder in which to place the your spotify data.

You can create a first file named `docker-compose.yml` and put the following content in it

export const response = await fetch("https://raw.githubusercontent.com/gamingdy/My-HomeLab/main/spotify/docker-compose.yml");
export const content = await response.text();

<Code code={content} lang="yml" title="docker-compose.yml" />

After that you need to configuration your environnement variable, you can see [here](https://github.com/Yooooomi/your_spotify?tab=readme-ov-file#creating-the-spotify-application) how to create a spotify application

export const responseClientEnv = await fetch("https://raw.githubusercontent.com/gamingdy/My-HomeLab/main/spotify/client.env");
export const contentClientEnv = await responseClientEnv.text();

<Code code={contentClientEnv} lang="txt" title="client.env" />

export const responseServerEnv = await fetch("https://raw.githubusercontent.com/gamingdy/My-HomeLab/main/spotify/server.env");
export const contentServerEnv = await responseServerEnv.text();

<Code code={contentServerEnv} lang="txt" title="server.env" />

Once the configuration is complete, you can run the following command

```bash
docker compose up -d
```
